CREATE TABLE YTBL_MODULOPROMOPRIN(
    ID NUMBER PRIMARY KEY,
    IDREGISTRO VARCHAR2(200),
    FGENERACIONREGISTRO DATE,
    NOMBREPROMOCION VARCHAR2(200),
    FINICIOCONTRATACION DATE,
    FFINCONTRATACION DATE,
    FFINANTICIPADACONTRATACION DATE,
    ISVALID VARCHAR2(3),
    PRODUCTOID NUMBER,
    VARIANTID NUMBER,
    REDID NUMBER,
    CANALID NUMBER,
    BURO VARCHAR2(50),
    EXCEPCION VARCHAR2(200),
    FORMADEPAGO VARCHAR2(35),
    ENTIDADBANCARIA VARCHAR2(530),
    EMISORTARJETA VARCHAR2(530),
    CIUDADES VARCHAR2(530),
    SECTORES CLOB,
    SUBSECTORES CLOB,
    MESINICIOPROMOCION NUMBER,
    MESFINPROMOCION VARCHAR2(7),
    FFINPROMOCION DATE,
    DIASGOZADOS VARCHAR2(3),
    PRECIOPROMOCIONAL NUMBER,
    PRECIOREFERENCIAL NUMBER,
    UPGRADES CLOB,
    FCADUCIDADANTICIPADAREGISTRO DATE
);

CREATE SEQUENCE YMODPROP_IN_ID START WITH 1 INCREMENT BY 1 NOCACHE NOCYCLE;

CREATE OR REPLACE TRIGGER YMODPROP_IN_ID_TRG
BEFORE INSERT ON YTBL_MODULOPROMOPRIN
FOR EACH ROW
BEGIN
    IF :NEW.ID IS NULL THEN
        SELECT YMODPROP_INC_ID.NEXTVAL INTO :NEW.ID FROM dual;
    END IF;
END;
